def sanitize_filename(name_part: str) -> str:
    """Cleans a string to make it suitable as a filename component.
    
    Removes or replaces unsafe/inappropriate characters from the input string to create a
    valid filename. The resulting string contains only alphanumeric characters, spaces,
    dots, underscores, and hyphens. Spaces are replaced with underscores, and the
    length is limited to 60 characters.

    Args:
        name_part: Input string to be sanitized. Will be converted to string if not already.

    Returns:
        Sanitized string suitable for use as a filename component. Returns an empty string
        if the input is empty or falsy.
    """
    if not name_part:
        return ""
    name_part = str(name_part) # 确保是字符串
    # 替换路径分隔符，移除不期望的字符
    name_part = name_part.replace(os.path.sep, '_')
    # 只保留字母数字、空格、点、下划线、中横线
    valid_chars = "".join(c for c in name_part if c.isalnum() or c in (' ', '.', '_', '-')).strip()
    # 避免文件名过长，并替换空格为下划线
    return valid_chars[:60].replace(' ', '_')
